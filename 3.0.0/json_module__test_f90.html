NAME

    json_check_for_errors

DESCRIPTION

    Retrieve error code from the module.
    This should be called after json_parse to check for errors.
    If an error is thrown, before using the module again, json_initialize
    should be called to clean up before it is used again.

EXAMPLE

    type(json_file) :: json
    logical :: status_ok
    character(len=:),allocatable :: error_msg
    call json%load_file(filename='myfile.json')
    call json_check_for_errors(status_ok, error_msg)
    if (.not. status_ok) then
        write(*,*) 'Error: '//error_msg
        call json_clear_exceptions()
        call json%destroy()
    end if

NAME

    json_get_array

DESCRIPTION

    This routine calls the user-supplied array_callback subroutine
        for each element in the array.
    Note: for integer, double, logical, and character arrays,
        a higher-level routine is provided (see json_get), so
        this routine does not have to be used for those cases.

SOURCE

3506 
3507     implicit none
3508 
3509     type(json_value),pointer,intent(in)  :: this
3510     character(len=*),intent(in),optional :: path
3511     procedure(array_callback_func)       :: array_callback
3512     logical,intent(out),optional         :: found
3513 
3514     type(json_value), pointer :: element,p
3515     integer :: i, count
3516 
3517     if (.not. exception_thrown) then
3518 
3519         nullify(p)
3520 
3521         ! resolve the path to the value
3522         if (present(path)) then
3523             call json_get_by_path(this=this, path=path, p=p)
3524         else
3525             p => this
3526         end if
3527 
3528         if (.not. associated(p)) then
3529 
3530             call throw_exception('Error in json_get_array:'//&
3531                                  ' Unable to resolve path: '//trim(path))
3532 
3533         else
3534 
3535             !associate (d => p%data)
3536                 select case (p%data%var_type)
3537                 case (json_array)
3538                     count = json_value_count(p)
3539                     do i = 1, count
3540                         call json_value_get(p, i, element)
3541                         call array_callback(element, i, count)
3542                     end do
3543                 case default
3544                     call throw_exception('Error in json_get_array:'//&
3545                                          ' Resolved value is not an array. '//&
3546                                          trim(path))
3547                 end select
3548             !end associate
3549 
3550             !cleanup:
3551             if (associated(p))       nullify(p)
3552             if (associated(element)) nullify(element)
3553 
3554         end if
3555 
3556         if (exception_thrown) then
3557             if (present(found)) then
3558                 found = .false.
3559                 call json_clear_exceptions()
3560             end if
3561         else
3562             if (present(found)) found = .true.
3563         end if

2589     subroutine json_get_by_path(this, path, p, found)
2590 
2591     implicit none
2592 
2593     type(json_value),pointer,intent(in)     :: this
2594     character(len=*),intent(in)             :: path
2595     type(json_value),pointer,intent(out)    :: p
2596     logical,intent(out),optional            :: found
2597 
2598     integer :: i, length, child_i
2599     character(len=1) :: c
2600     logical :: array
2601     type(json_value),pointer :: tmp
2602 
2603     if (.not. exception_thrown) then
2604 
2605         nullify(p)
2606 
2607         ! default to assuming relative to this
2608         p => this
2609 
2610         child_i = 1
2611 
2612         array = .false.
2613 
2614         length = len_trim(path)
2615 
2616         do i=1, length
2617 
2618             c = path(i:i)
2619 
2620             select case (c)
2621             case ('$')
2622 
2623                 ! root
2624                 do while (associated (p % parent))
2625                     p => p % parent
2626                 end do
2627                 child_i = i + 1
2628 
2629             case ('@')
2630 
2631                 ! this
2632                 p => this
2633                 child_i = i + 1
2634 
2635             case ('.')
2636 
2637                 ! get child member from p
2638                 if (child_i < i) then
2639                     nullify(tmp)
2640                     call json_value_get(p, path(child_i:i-1), tmp)
2641                     p => tmp
2642                     nullify(tmp)
2643                 else
2644                     child_i = i + 1
2645                     cycle
2646                 end if
2647 
2648                 if (.not. associated(p)) then
2649                     call throw_exception('Error in json_get_by_path:'//&
2650                                          ' Error getting child member.')
2651                     exit
2652                 end if
2653 
2654                 child_i = i+1
2655 
2656             case ('[','(')
2657 
2658                 !....Modified to allow for 'var[3]' style syntax
2659                 !Note: jmozmoz/fson has a slightly different version of this...
2660 
2661                 ! start looking for the array element index
2662                 array = .true.
2663 
2664                 ! get child member from p
2665                 if (child_i < i) then
2666                     nullify(tmp)
2667                     call json_value_get(p, path(child_i:i-1), tmp)
2668                     p => tmp
2669                     nullify(tmp)
2670                 else
2671                     child_i = i + 1
2672                     cycle
2673                 end if
2674                 if (.not. associated(p)) then
2675                     call throw_exception('Error in json_get_by_path:'//&
2676                                          ' Error getting array element')
2677                     exit
2678                 end if
2679                 child_i = i + 1
2680 
2681             case (']',')')
2682 
2683                 if (.not.array) then
2684                     call throw_exception('Error in json_get_by_path: Unexpected ]')
2685                     exit
2686                 end if
2687                 array = .false.
2688                 child_i = string_to_integer(path(child_i:i-1))
2689 
2690                 nullify(tmp)
2691                 call json_value_get(p, child_i, tmp)
2692                 p => tmp
2693                 nullify(tmp)
2694 
2695                 child_i= i + 1
2696 
2697             end select
2698 
2699         end do
2700 
2701         if (exception_thrown) then
2702

2704                 found = .false.
2705                 call json_clear_exceptions()
2706             end if
2707 
2708         else
2709 
2710             ! grab the last child if present in the path
2711             if (child_i <= length) then
2712                 nullify(tmp)
2713                 call json_value_get(p, path(child_i:i-1), tmp)
2714                 p => tmp
2715                 nullify(tmp)
2716             end if
2717             if (associated(p)) then
2718                 if (present(found)) found = .true.    !everything seems to be ok
2719             else
2720                 call throw_exception('Error in json_get_by_path:'//&
2721                                      ' variable not found: '//trim(path))
2722                 if (present(found)) then
2723                     found = .false.
2724                     call json_clear_exceptions()
2725                 end if
2726             end if
2727             
2728         end if
2729 
2730     else
2731         if (present(found)) found = .false.
2732     end if
2733 
2734     end subroutine json_get_by_path

NAME

    json_value_destroy

DESCRIPTION

    Remove a json_value (and all its children) 
      from a linked-list structure, preserving the rest of the structure.
    If destroy is not present, it is also destroyed.
    If destroy is present and true, it is destroyed.
    If destroy is present and false, it is not destroyed.

EXAMPLE

    !to extract an object from one json structure, and add it to another:
    type(json_value),pointer :: json1,json2,p
    logical :: found
    [create and populate json1 and json2]
    call json_get(json1,'name',p,found)  ! get pointer to name element of json1
    call json_remove(p,destroy=.false.)  ! remove it from json1 (don't destroy)
    call json_value_add(json2,p)         ! add it to json2

    !to remove an object from a json structure (and destroy it)
    type(json_value),pointer :: json1,p
    logical :: found
    [create and populate json1]
    call json_get(json1,'name',p,found)  ! get pointer to name element of json1
    call json_remove(p)                  ! remove and destroy it
