    call json%get('var.i',ival,found)

    call json%get('var.c',cval,found)
    call json%destroy()

3313 
3314     implicit none
3315 
3316     type(json_value),pointer,intent(in)  :: this
3317     character(len=*),intent(in),optional :: path
3318     procedure(array_callback_func)       :: array_callback
3319     logical,intent(out),optional         :: found
3320 
3321     type(json_value), pointer :: element,p
3322     integer :: i, count
3323 
3324     if (.not. exception_thrown) then
3325 
3326         nullify(p)
3327 
3328         ! resolve the path to the value
3329         if (present(path)) then
3330             call json_get_by_path(this=this, path=path, p=p)
3331         else
3332             p => this
3333         end if
3334

3336

3341

3347                         call json_value_get(p, i, element)

3355

3359

3364                 found = .false.
3365                 call json_clear_exceptions()
