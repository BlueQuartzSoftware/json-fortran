3031

    json_get_array

DESCRIPTION

    Get an array from an json_value.
    This routine calls the user-supplied array_callback subroutine
        for each element in the array.

SOURCE

2841     subroutine json_get_array(this, path, array_callback, found)
2842 
2843     implicit none
2844 
2845     type(json_value),pointer,intent(in)  :: this
2846     character(len=*),intent(in),optional :: path
2847     procedure(array_callback_func)       :: array_callback
2848     logical,intent(out),optional         :: found
2849 
2850     type(json_value), pointer :: element,p
2851     integer :: i, count
2852 
2853     if (.not. exception_thrown) then
2854 
2855         nullify(p)
2856 
2857         ! resolve the path to the value
2858         if (present(path)) then
2859             call json_get_by_path(this=this, path=path, p=p)
2860         else
2861             p => this
2862         end if
2863 
2864         if (.not.associated(p)) then
2865 
2866             call throw_exception('Error in json_get_array: Unable to resolve path: '//trim(path))
2867 
2868         else
2869 
2870             !associate (d => p%data)
2871                 select case (p%data%var_type)
2872                 case (json_array)
2873                     count = json_value_count(p)
2874                     do i = 1, count
2875                         call json_value_get(p, i, element)
2876                         call array_callback(element, i, count)
2877                     end do
2878                 case default
2879                     call throw_exception('Error in json_get_array: Resolved value is not an array. '//trim(path))
2880                 end select
2881             !end associate
2882 
2883             !cleanup:
2884             if (associated(p))       nullify(p)
2885             if (associated(element)) nullify(element)
2886 
2887         end if
2888 
2889         if (exception_thrown) then
2890             if (present(found)) then
2891                 found = .false.
2892                 call json_clear_exceptions()
2893             end if
2894         else
2895             if (present(found)) found = .true.
2896         end if
2897
