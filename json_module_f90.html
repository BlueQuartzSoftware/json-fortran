    json_get_array

DESCRIPTION

    Get an array from an json_value.
    This routine calls the user-supplied array_callback subroutine
        for each element in the array.

SOURCE

2842     subroutine json_get_array(this, path, array_callback, found)
2843 
2844     implicit none
2845 
2846     type(json_value),pointer,intent(in)  :: this
2847     character(len=*),intent(in),optional :: path
2848     procedure(array_callback_func)       :: array_callback
2849     logical,intent(out),optional         :: found
2850 
2851     type(json_value), pointer :: element,p
2852     integer :: i, count
2853 
2854     if (.not. exception_thrown) then
2855 
2856         nullify(p)
2857 
2858         ! resolve the path to the value
2859         if (present(path)) then
2860             call json_get_by_path(this=this, path=path, p=p)
2861         else
2862             p => this
2863         end if
2864 
2865         if (.not.associated(p)) then
2866 
2867             call throw_exception('Error in json_get_array: Unable to resolve path: '//trim(path))
2868 
2869         else
2870

2872                 select case (p%data%var_type)
2873                 case (json_array)
2874                     count = json_value_count(p)
2875                     do i = 1, count
2876                         call json_value_get(p, i, element)
2877                         call array_callback(element, i, count)
2878                     end do
2879                 case default
2880                     call throw_exception('Error in json_get_array: Resolved value is not an array. '//trim(path))
2881                 end select
2882             !end associate
2883 
2884             !cleanup:
2885             if (associated(p))       nullify(p)
2886             if (associated(element)) nullify(element)
2887 
2888         end if
2889 
2890         if (exception_thrown) then
2891             if (present(found)) then
